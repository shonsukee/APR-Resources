import os
import uuid
import time
import hashlib
import hmac
import base64

class SwitchBot:
    BASE_URL        = 'https://api.switch-bot.com'
    BASE_PATH       = os.path.dirname(os.path.abspath(__file__))
    TOKEN_FILE      = os.path.join(BASE_PATH, 'token.txt')
    SECRET_KEY      = os.path.join(BASE_PATH, 'secret.txt')  # Assuming there's a file for secret key

    def __init__(self):
        with open(SwitchBot.TOKEN_FILE, 'r') as f:
            self.token = f.read().strip()
        with open(SwitchBot.SECRET_KEY, 'r') as f:
            self.secret = f.read().strip()

    def get_headers(self):
        nonce = str(uuid.uuid4())
        t = str(int(round(time.time() * 1000)))
        string_to_sign = '{}{}{}'.format(self.token, t, nonce)
        
        sign = base64.b64encode(
            hmac.new(self.secret.encode('utf-8'), msg=string_to_sign.encode('utf-8'), digestmod=hashlib.sha256).digest()
        ).decode('utf-8')

        return {
            'Authorization': self.token,
            'Content-Type': 'application/json',
            't': t,
            'sign': sign,
            'nonce': nonce
        }

    def get_url(self, *pathes):
        url = SwitchBot.BASE_URL + '/v1.1'
        for path in pathes:
            url += '/' + path
        return url

    def do_get(self, url, data, headers={}):
        headers.update(self.get_headers())
        req = Request(url, data, headers)
        res = urlopen(req)
        return res.read().decode()

import java.net.http.HttpResponse;
import java.net.http.HttpResponse.BodyHandlers;
import java.time.Instant;
import java.util.Base64;
import java.util.UUID;

import javax.crypto.Mac;
import javax.crypto.spec.SecretKeySpec;

public class Main {
    public static void main(String[] args) throws Exception {
    	
      String token = args[0];
      String secret = args[1];
      String nonce = UUID.randomUUID().toString();
      String time= "" + Instant.now().toEpochMilli();
      String data = token + time + nonce;
      
      SecretKeySpec secretKeySpec = new SecretKeySpec(secret.getBytes("UTF-8"), "HmacSHA256");
      Mac mac = Mac.getInstance("HmacSHA256");
      mac.init(secretKeySpec);
      String signature = new String(Base64.getEncoder().encode(mac.doFinal(data.getBytes("UTF-8"))));      
      
      HttpRequest getDevices = HttpRequest.newBuilder()
      .uri(new URI("https://api.switch-bot.com/v1.1/devices"))
      .header("Authorization", token)
      .header("sign", signature)
      .header("nonce", nonce)
      .header("t", time)
      .GET()
      .build();
      
      HttpResponse<String> response = HttpClient.newBuilder().build().send(getDevices, BodyHandlers.ofString());
      
      System.out.println(response.body());
    }
